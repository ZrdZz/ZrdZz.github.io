### 数组中出现次数超过一半的数字

普通解法
```
function MoreThanHalfNum_Solution(numbers){
    var len = numbers.length;
    for(let i = 0; i < len / 2; i++){
        var count = 1;
        for(let j = 0; j < len; j++){
            if(j === i){
                continue;
            }
            if(numbers[i] === numbers[j]){
                count++
            }
        }        
        if(count > len / 2){
            return numbers[i]
        }else{
            continue
        }
    }
    return 0
}
```

数组排序后,若有符合条件的数,一定是中间的数
```
function MoreThanHalfNum_Solution(numbers){
    numbers.sort();
    var len = numbers.length,
        middle = numbers[Math.floor(len / 2)],
        count = 0;
    
    for(let i = 0; i < len; i++){
        if(numbers[i] === middle){
            count++
        }
    }
    return count > len / 2 ? middle : 0
}
```
